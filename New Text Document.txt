<script>
    $(document).ready(function(){
      
      $(".add-row").click(function(){
            $("#table-new-members tbody").find('input[name="record"]').each(function(){
            	if($(this).is(":checked")){
                  $("#table-members-in-group tbody").append($(this).parents("tr"));
                }
            });
        });
        
        // Find and remove selected table rows
        $(".delete-row").click(function(){
            $("#table-members-in-group tbody").find('input[name="record"]').each(function(){
            	if($(this).is(":checked")){
                    $(this).parents("tr").remove();
                }
            });
        });
    });    
</script>
